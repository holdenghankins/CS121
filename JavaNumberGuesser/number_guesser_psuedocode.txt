public static int askForVersion() {
    Scanner scanner = new scanner(System.out);
    System.out.println("Which version:\n(0) exit\n(1) User Guess\n(2) Computer Guess");
    int version = scanner.nextInt;
    if (version != 0 && version != 1 && version != 2 ) {
        System.out.println("Invalid Input")
        return askForVersion(); 
    }
    else {
        return version;
    }
}

public void computerGuess(int numberToGuess, int low, int high) {
    int guess = (low + high) / 2;
    if (guess == low) {
        guess++;
    }

    System.out.println("I guess " + guess);
    char input == getHighLowCorrect()

    }
    return;
}

public char getHighLowCorrect() {
    Scanner scanner = new Scanner(System.in);
    System.out.println("Too (H)igh, too (L)ow, or (C)orrect?");
    char input = scanner.nextChar().toLowerCase();
    if (input != 'h' && input != 'l' && input != 'c') {
        System.out.println("Invalid Input");
        return getHighLowCorrect();
    }
    return input;
}
